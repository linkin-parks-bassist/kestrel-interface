39uint32_t m_encode_dsp_block_instr_type_a(m_block *block)
40{
41 if (!block) return 0;
42
43 return place_bits( 5, 0, block->instr)
44 | place_bits( 9, 6, block->arg_a.addr) | ((!!(block->arg_a.type == M_ASM_ARG_EXPR)) << 10)
45 | place_bits(14, 11, block->arg_b.addr) | ((!!(block->arg_b.type == M_ASM_ARG_EXPR)) << 15)
46 | place_bits(19, 16, block->arg_c.addr) | ((!!(block->arg_c.type == M_ASM_ARG_EXPR)) << 20)
47 | place_bits(24, 21, block->dest)
48 | place_bits(29, 25, block->shift) | ((!!block->saturate_disable) << 30) | ((block->shift == 255) << 31);
49}

51uint32_t m_encode_dsp_block_instr_type_b(m_block *block, int res_handle)
52{
53 return place_bits(5, 0, block->instr) | (1 << 5)
54 | place_bits( 9, 6, block->arg_a.addr) | ((!!(block->arg_a.type == M_ASM_ARG_EXPR)) << 10)
55 | place_bits(14, 11, block->arg_b.addr) | ((!!(block->arg_b.type == M_ASM_ARG_EXPR)) << 15)
56 | place_bits(19, 16, block->dest)
57 | place_bits(31, 20, res_handle);
58}

401void print_instruction_format_a(uint32_t instr)
402{
403 int opcode = range_bits(instr, 5, 0);
404
405 int src_a = range_bits(instr, 4, 6);
406 int src_a_reg = !!(instr & (1 << 10));
407
408 int src_b = range_bits(instr, 4, 11);
409 int src_b_reg = !!(instr & (1 << 15));
410
411 int src_c = range_bits(instr, 4, 16);
412 int src_c_reg = !!(instr & (1 << 20));
413
414 int dest = range_bits(instr, 4, 21);
415 int shift = range_bits(instr, 5, 25);
416 int sat = !!(instr & (1 << 30));
417 int no_shift = !!(instr & (1 << 31));
418
419 M_PRINTF_("%s %s%d %s%d %s%d, c%d (%d%s)",
420 m_block_opcode_to_name(opcode),
421 src_a_reg ? "r" : "c", src_a,
422 src_b_reg ? "r" : "c", src_b,
423 src_c_reg ? "r" : "c", src_c, dest,
424 shift, sat ? ", unsat" : "");

463int m_fpga_batch_print(m_fpga_transfer_batch seq)
464{
465 int n = seq.len;
466
467 M_PRINTF_("Reading out FPGA transfer batch %p (length %d)\n", seq.buf, n);
468
469 if (!seq.buf)
470 {
471 M_PRINTF_("Buffer is NULL!\n");
472 }
473
474 if (n < 1)
475 {
476 M_PRINTF_("Batch has no bytes!\n");
477 }
478
479 int i = 0;
480
481 int state = 0;
482 int ret_state;
483 int skip = 0;
484
485 int shift = 0;
486
487 uint8_t byte;
488
489 int ctr = 0;
490 int ctr_2 = 0;
491
492 uint8_t reg_no = 0;
493 m_fpga_block_addr_t block = 0;
494 uint32_t value = 0;
495 int32_t signed_val;
496 uint32_t instruction = 0;
497
498 while (i < n)
499 {
500 byte = seq.buf[i];
501
502 M_PRINTF_("\tByte %s%d: 0x%02X. ", (n > 9 && i < 10) ? " " : "", i, byte);
503
504 switch (state)
505 {
506 case 0: // expecting a command
507 switch (byte)
508 {
509 case COMMAND_BEGIN_PROGRAM:
510 M_PRINTF_("Command BEGIN_PROGRAM");
511 break;
512
513 case COMMAND_WRITE_BLOCK_INSTR:
514 M_PRINTF_("Command WRITE_BLOCK_INSTR");
515 state = 1;
516 ret_state = 2;
517 ctr = 0;
518 instruction = 0;
519 break;
520
521 case COMMAND_WRITE_BLOCK_REG_0:
522 M_PRINTF_("Command WRITE_BLOCK_REG_0");
523 state = 1;
524 ret_state = 4;
525 value = 0;
526 ctr = 0;
527 break;
528
529 case COMMAND_WRITE_BLOCK_REG_1:
530 M_PRINTF_("Command WRITE_BLOCK_REG_1");
531 state = 1;
532 ret_state = 4;
533 value = 0;
534 ctr = 0;
535 break;
536
537 case COMMAND_ALLOC_DELAY:
538 M_PRINTF_("Command ALLOC_DELAY");
539 state = 5;
540 value = 0;
541 ctr = 0;
542 ctr_2 = 0;
543 shift = 15;
544 break;
545
546 case COMMAND_END_PROGRAM:
547 M_PRINTF_("Command END_PROGRAM");
548 break;
549
550 case COMMAND_UPDATE_BLOCK_REG_0:
551 M_PRINTF_("Command UPDATE_BLOCK_REG_0");
552 state = 1;
553 ret_state = 4;
554 value = 0;
555 ctr = 0;
556 break;
557
558 case COMMAND_UPDATE_BLOCK_REG_1:
559 M_PRINTF_("Command UPDATE_BLOCK_REG_1");
560 state = 1;
561 ret_state = 4;
562 value = 0;
563 ctr = 0;
564 break;
565
566 case COMMAND_COMMIT_REG_UPDATES:
567 M_PRINTF_("Command COMMIT_REG_UPDATES");
568 break;
569
570 case COMMAND_SET_INPUT_GAIN:
571 M_PRINTF_("Command SET_INPUT_GAIN");
572 state = 4;
573 ret_state = 0;
574 value = 0;
575 ctr = 0;
576 shift = 4;
577 break;
578
579 case COMMAND_SET_OUTPUT_GAIN:
580 M_PRINTF_("Command SET_OUTPUT_GAIN");
581 state = 4;
582 ret_state = 0;
583 value = 0;
584 ctr = 0;
585 shift = 4;
586 break;
587
588 case COMMAND_ALLOC_FILTER:
589 M_PRINTF_("Command COMMAND_ALLOC_FILTER");
590 state = 6;
591 ret_state = 0;
592 value = 0;
593 ctr = 0;
594 ctr_2 = 0;
595 shift = 0;
596 break;
597
598 case COMMAND_WRITE_FILTER_COEF:
599 M_PRINTF_("Command COMMAND_WRITE_FILTER_COEF");
600 state = 7;
601 ret_state = 0;
602 value = 0;
603 ctr = 0;
604 ctr_2 = 0;
605 shift = 0;
606 break;
607
608 }
609 break;
610
611 case 1: // expecting block number
612 block = (block << 8) | byte;
613
614 if (ctr == M_FPGA_BLOCK_ADDR_BYTES - 1)
615 {
616 M_PRINTF_("Block number %d", byte);
617 state = ret_state;
618 ctr = 0;
619 }
620 else
621 {
622 ctr++;
623 }
624
625 break;
626
627 case 2: // expecting instruction
628 if (ctr == 3)
629 {
630 state = 0;
631 instruction = (instruction << 8) | byte;
632 //m_printf("Word: %s; ", binary_print_32(instruction));
633 print_instruction(instruction);
634 if (m_fpga_block_opcode_format(instruction & IBM(5)))
635 shift = 0;
636 else
637 shift = range_bits(instruction, 5, 25);
638 }
639 else
640 {
641 instruction = (instruction << 8) | byte;
642 ctr++;
643 }
644 break;
645
646 case 3: // expecting register number then register value
647 M_PRINTF_("Register %d", byte);
648 state = 4;
649 break;
650
651 case 4: // expecting a value
652 if (ctr == M_FPGA_DATA_BYTES - 1)
653 {
654 state = 0;
655
656 value = (value << 8) | byte;
657 M_PRINTF_("Value: %s = %d = %f (in q%d.%d)", binary_print_16(value), value, (float)value / (powf(2.0, 15 - shift)), 1 + shift, 15 - shift);
658 }
659 else
660 {
661 value = (value << 8) | byte;
662 ctr++;
663 }
664 break;
665
666 case 5: // expecting two 24-bit values. lol!
667 if (ctr == 0)
668 {
669 value = byte;
670 ctr++;
671 }
672 else if (ctr == 2)
673 {
674 value = (value << 8) | byte;
675 M_PRINTF_("%s: %s = 0x%06x = %.02f", ctr_2 ? "Delay" : "Size", binary_print_24(value), value, (float)((uint32_t)value) / (powf(2.0, (15 - shift))));
676
677 ctr = 0;
678
679 if (ctr_2 == 1)
680 {
681 state = 0;
682 }
683 else
684 {
685 shift = 7;
686 ctr_2 = 1;
687 }
688 }
689 else
690 {
691 value = (value << 8) | byte;
692 ctr++;
693 }
694 break;
695
696 case 6:
697 if (ctr_2 == 0)
698 {
699 M_PRINTF_("Format: q%d.%d", 1 + byte, 17 - byte);
700 shift = byte;
701 ctr_2++;
702 }
703 else if (ctr_2 == 1)
704 {
705 if (ctr == 0)
706 {
707 value = byte;
708 ctr++;
709 }
710 else
711 {
712 value = (value << 8) | byte;
713 M_PRINTF_("Feed-forward: %d", value);
714
715 ctr_2++;
716 ctr = 0;
717 }
718 }
719 else
720 {
721 if (ctr == 0)
722 {
723 value = byte;
724 ctr++;
725 }
726 else
727 {
728 value = (value << 8) | byte;
729 M_PRINTF_("Feed-back: %d", value);
730
731 state = 0;
732 }
733 }
734
735 break;
736
737 case 7:
738 if (ctr_2 == 0)
739 {
740 M_PRINTF_("Filter %d", byte);
741 ctr_2++;
742 ctr = 0;
