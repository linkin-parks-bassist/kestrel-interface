442int kest_scope_propagate_updates(kest_scope *scope)
443{
444 if (!scope)
445 return ERR_NULL_PTR;
446
447 kest_scope_entry *entry = NULL;
448 kest_scope_entry *ref_entry = NULL;
449 size_t n = kest_scope_count(scope);
450 int ret_val;
451
452 string_list names;
453
454 if (!n)
455 return NO_ERROR;
456
457 char_ptr_list_init(&names);
458
459 for (size_t i = 0; i < n; i++)
460 {
461 entry = kest_scope_index(scope, i);
462
463 if (!entry)
464 continue;
465
466 if (entry->type == KEST_SCOPE_ENTRY_TYPE_PARAM)
467 {
468 entry->updated = entry->val.param ? entry->val.param->updated : 0;
469 }
470 else if (entry->type == KEST_SCOPE_ENTRY_TYPE_SETTING)
471 {
472 entry->updated = entry->val.setting ? entry->val.setting->updated : 0;
473 }
474 }
475
476 for (size_t i = 0; i < n; i++)
477 {
478 entry = kest_scope_index(scope, i);
479
480 if (entry && entry->type == KEST_SCOPE_ENTRY_TYPE_EXPR)
481 {
482 ret_val = kest_expression_get_references(entry->val.expr, &names);
483
484 if (ret_val != NO_ERROR)
485 {
486 char_ptr_list_destroy(&names);
487 return ret_val;
488 }
489
490 for (int j = 0; j < names.count; j++)
491 {
492 ref_entry = kest_scope_lookup(scope, names.entries[j]);
493
494 if (ref_entry && ref_entry->updated)
495 {
496 entry->updated = 1;
497 break;
498 }
499 }
500
501 char_ptr_list_drain(&names);
502 }
503 }
504
505 char_ptr_list_destroy(&names);
506
507 return NO_ERROR;
508}
