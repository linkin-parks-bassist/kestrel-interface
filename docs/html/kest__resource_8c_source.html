143{
144 int next_filter_handle = 0;

146 int next_mem_handle = 0;

417int kest_lfo_evaluate_rec(kest_lfo *lfo, kest_scope *scope, float *dest, int depth)
418{
419 if (!lfo || !dest)
420 return ERR_NULL_PTR;
421
422 float result = 0.0f;
423 float tmp = 0.0f;
424
425 float min = 0.0f;
426 float max = 0.0f;
427 float center = 0.0f;
428 float amp = 0.0f;
429
430 float freq = kest_expression_evaluate_rec(lfo->frequency, scope, depth + 1);
431
432 KEST_PRINTF("freq = %s = %f\n", kest_expression_to_string(lfo->frequency), freq);
433
434 int64_t time_ms = kest_system_time_ms();
435
436 float t = lfo->prev_t + 2.0f * M_PI * freq * ((float)(time_ms - lfo->prev_ms) * 0.001);
437
438 while (t > 2.0f * M_PI)
439 t -= 2.0f * M_PI;
440
441 KEST_PRINTF("lfo->mode = %d, lfo->scale = %d\n", lfo->mode, lfo->scale);
442
443 switch (lfo->mode)
444 {
445 case KEST_LFO_MODE_MIN_MAX:
446
447 min = kest_expression_evaluate_rec(lfo->min, scope, depth + 1);
448 max = kest_expression_evaluate_rec(lfo->max, scope, depth + 1);
449
450 if (min > max)
451 {
452 tmp = max;
453 max = min;
454 min = tmp;
455 }
456
457 center = 0.5 * (min + max);
458 amp = 0.5 * (max - min);
459
460 break;
461
462 case KEST_LFO_MODE_CENTER_AMP:
463
464 center = kest_expression_evaluate_rec(lfo->center, scope, depth + 1);
465 amp = kest_expression_evaluate_rec(lfo->amplitude, scope, depth + 1);
466
467 min = center - amp;
468 max = center + amp;
469
470 break;
471 }
472
473 if (lfo->scale == KEST_LFO_SCALE_LOG)
474 {
475 min = min < 0 ? -FLT_MAX : log(min);
476 max = max < 0 ? -FLT_MAX : log(max);
477
478 center = 0.5 * (min + max);
479 amp = 0.5 * (max - min);
480 }
481
482 KEST_PRINTF("result = amp * sin(t) + center = %f + sin(%f) + %f = %f * %f + %f = %f\n",
483 amp, t, center, amp, sin(t), center, amp * sin(t) + center);
484
485 result = amp * sin(t) + center;
486
487 if (lfo->scale == KEST_LFO_SCALE_LOG)
488 result = expf(result);
489
490 lfo->prev_t = t;
491 lfo->prev_ms = time_ms;
492 *dest = result;
493
494 return NO_ERROR;
495}
