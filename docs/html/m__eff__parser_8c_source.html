379int m_parser_lineize_content(m_eff_parsing_state *ps)
380{
381 if (!ps)
382 return ERR_NULL_PTR;
383
384 if (!ps->content)
385 return ERR_BAD_ARGS;
386
387 if (!ps->n_lines)
388 return NO_ERROR;
389
390 int line = 1;
391 ps->lines = m_parser_alloc(sizeof(char*) * ps->n_lines);
392
393 if (!ps->lines)
394 return ERR_ALLOC_FAIL;
395
396 for (int i = 0; i < ps->n_lines; i++)
397 ps->lines[i] = NULL;
398
399 ps->lines[0] = ps->content;
400
401 int line_start = 0;
402
403 for (int i = 0; i < ps->file_size && line < ps->n_lines; i++)
404 {
405 if (line_start)
406 ps->lines[line++] = &ps->content[i];
407
408 line_start = 0;
409
410 if (ps->content[i] == '\n')
411 {
412 ps->content[i] = 0;
413 line_start = 1;
414 }
415 }
416
417 // It may be the case that, if the final line is just "\n", the loop will end one
418 // line early. Then, there is a risk of segfault if one tries to deref ps->lines[n_lines-1]
419 // therefore, decrement ps->n_lines to prevent this.
420 ps->n_lines = line;
421
422 return NO_ERROR;
423}

560int m_parser_format_offending_section(char *line, int index, int length, char *buf, int buf_len, char *colour)
561{
562 if (!line || !buf)
563 return ERR_NULL_PTR;
564
565 int i, j, buf_pos;
566
567 buf_pos = 0;
568
569 for (i = 0; i < PR_LINE_INDENT; i++)
570 buf[buf_pos++] = ' ';
571
572 for (i = 0; i <= index && buf_pos + 1 < buf_len && line[i] != 0; i++)
573 buf[buf_pos++] = line[i];
574
575 if (colour)
576 {
577 for (j = 0; err_colour[j] != 0 && buf_pos + 1 < buf_len; j++)
578 buf[buf_pos++] = colour[j];
579 }
580
581 for (j = 0; j < length && buf_pos < buf_len; j++)
582 buf[buf_pos++] = line[i++];
583
584 if (colour)
585 {
586 for (j = 0; reset_colour[j] != 0 && buf_pos + 1 < buf_len; j++)
587 buf[buf_pos++] = reset_colour[j];
588 }
589
590 for (j = 0; line[i] != 0 && buf_pos + 1 < buf_len; j++)
591 buf[buf_pos++] = line[i++];
592
593 if (buf_pos + index + length / 2 + 1 + PR_LINE_INDENT >= buf_len)
594 {
595 buf[buf_pos < buf_len ? buf_pos : buf_len - 1] = 0;
596 return NO_ERROR;
597 }
598 else
599 {
600 buf[buf_pos++] = '\n';
601 }
602
603 for (int i = 0; i <= index + length / 2 + PR_LINE_INDENT - 1 && buf_pos + 1 < M_PARSER_PRINT_BUFLEN; i++)
604 buf[buf_pos++] = '~';
605
606 buf[buf_pos++] = '^';
607 buf[buf_pos++] = 0;
608
609 return NO_ERROR;
610}
