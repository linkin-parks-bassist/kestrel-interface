144int m_profile_set_default_name_from_id(m_profile *profile)
145{
146 M_PRINTF("m_profile_set_default_name_from_id\n");
147 if (!profile)
148 return ERR_NULL_PTR;
149
150 M_PRINTF("ID = %d\n", profile->id);
151
152 // Compute the digits in the ID.
153 int id_digits;
154 int id_div = profile->id;
155
156 for (id_digits = 0; id_div || !id_digits; id_div = id_div / 10)
157 id_digits++;
158
159 if (profile->name)
160 m_free(profile->name);
161
162 profile->name = m_alloc(9 + id_digits);
163
164 if (!profile->name)
165 return ERR_ALLOC_FAIL;
166
167 sprintf(profile->name, "Profile %d", profile->id);
168
169 M_PRINTF("Resulting name: %s\n", profile->name);
170
171 return NO_ERROR;
