53int kest_fpga_txrx(uint8_t *tx, uint8_t *rx, size_t len)
54{
55 #ifdef KEST_FPGA_SIMULATED
56 if (rx)
57 {
58 for (int i = 0; i < len; i++)
59 rx[i] = 1;
60 }
61 return ERR_FEATURE_DISABLED;
62 #else
63 if (len == 0)
64 return 0;
65
66 uint8_t reply_buf[len];
67 if (!rx)
68 rx = reply_buf;
69
70 spi_transaction_t t = {
71 .length = len * 8,
72 .tx_buffer = tx,
73 .rx_buffer = rx,
74 };
75
76 esp_err_t err = spi_device_transmit(spi_handle, &t);
77
78 #ifdef PRINT_SPI_BYTES
79 char print_buf[len * 10 + 1];
80 int buf_index = 0;
81 for (int i = 0; i < len; i++)
82 {
83 sprintf(&print_buf[buf_index], "SEND 0x%02x\n", tx[i]);
84 buf_index += 10;
85 }
86 print_buf[buf_index] = 0;
87 KEST_PRINTF_FORCE("\nSPI send to FPGA (%d bytes):\n", len);
88 KEST_PRINTF_FORCE(print_buf);
89
90 KEST_PRINTF_FORCE("FPGA replied with\n");
91 buf_index = 0;
92 for (int i = 0; i < len; i++)
93 {
94 sprintf(&print_buf[buf_index], "RCVD 0x%02x\n", rx[i]);
95 buf_index += 10;
96 }
97 print_buf[buf_index] = 0;
98 KEST_PRINTF_FORCE(print_buf);
99 #endif
100
101 return (err == ESP_OK) ? NO_ERROR : ERR_SPI_FAIL;
102 #endif
103}

521int kest_fpga_decode_status_flags(kest_fpga_status_flags *flags, uint8_t byte)
522{
523 if (!flags)
524 return ERR_NULL_PTR;
525
526 flags->initialised = !!(byte & (1 << 0));
527 flags->listening = !!(byte & (1 << 1));
528 flags->timeout = !!(byte & (1 << 2));
529 flags->programming = !!(byte & (1 << 3));
530 flags->bad = !!(byte & (1 << 4));
531 flags->data_ready = !!(byte & (1 << 5));
532 flags->cmd_err = !!(byte & (1 << 6));
533 flags->swapping = !!(byte & (1 << 7));
534
535 return NO_ERROR;
536}

591int64_t kest_fpga_req_data(int req, int n_bytes, kest_fpga_status_flags *flags)
592{
593 if (!flags) return -1;
594
595 kest_fpga_send_byte_get_flags(COMMAND_READ, flags);
596 kest_fpga_send_byte_get_flags(req, flags);
597
598 if (flags->cmd_err)
599 return -2;
600
601 int tries = 16;
602
603 do {
604 kest_fpga_get_status_flags(flags);
605 if (flags->cmd_err) return -4;
606 #ifdef ENABLE_FREERTOS
607 if (!flags->data_ready)
608 vTaskDelay(1);
609 #endif
610 } while (!flags->data_ready && tries --> 0);
611
612 int64_t data = 0;
613
614 if (flags->data_ready)
615 {
616 kest_fpga_send_byte_get_flags(COMMAND_READOUT, flags);
617
618 if (flags->cmd_err)
619 return -6;
620
621 for (int i = 0; i < n_bytes; i++)
622 data = (data << 8) | kest_fpga_readout();
623 }
624 else
625 {
626 data = -5;
627 }
628
629 kest_fpga_send_byte_get_flags(COMMAND_CLEAR_CMD_ERR_FLAG, flags);
630
631 return data;
632}
