53int kest_fpga_txrx(uint8_t *tx, uint8_t *rx, size_t len)
54{
55 #ifdef KEST_FPGA_SIMULATED
56 if (rx)
57 {
58 for (int i = 0; i < len; i++)
59 rx[i] = 1;
60 }
61 return ERR_FEATURE_DISABLED;
62 #else
63 if (len == 0)
64 return 0;
65
66 uint8_t reply_buf[len];
67 if (!rx)
68 rx = reply_buf;
69
70 spi_transaction_t t = {
71 .length = len * 8,
72 .tx_buffer = tx,
73 .rx_buffer = rx,
74 };
75
76 esp_err_t err = spi_device_transmit(spi_handle, &t);
77
78 #ifdef PRINT_SPI_BYTES
79 char print_buf[len * 10 + 1];
80 int buf_index = 0;
81 for (int i = 0; i < len; i++)
82 {
83 sprintf(&print_buf[buf_index], "SEND 0x%02x\n", tx[i]);
84 buf_index += 10;
85 }
86 print_buf[buf_index] = 0;
87 KEST_PRINTF_FORCE("\nSPI send to FPGA (%d bytes):\n", len);
88 KEST_PRINTF_FORCE(print_buf);
89
90 KEST_PRINTF_FORCE("FPGA replied with\n");
91 buf_index = 0;
92 for (int i = 0; i < len; i++)
93 {
94 sprintf(&print_buf[buf_index], "RCVD 0x%02x\n", rx[i]);
95 buf_index += 10;
96 }
97 print_buf[buf_index] = 0;
98 KEST_PRINTF_FORCE(print_buf);
99 #endif
100
101 return (err == ESP_OK) ? NO_ERROR : ERR_SPI_FAIL;
102 #endif
103}

294int kest_fpga_batch_concat(kest_fpga_transfer_batch *seq, kest_fpga_transfer_batch *seq2)
295{
296 if (!seq || !seq2)
297 return ERR_NULL_PTR;
298
299 if (!seq2->buf)
300 return ERR_BAD_ARGS;
301
302 if (!seq->buf)
303 {
304 seq->buf = kest_alloc(seq2->len);
305
306 if (!seq->buf)
307 return ERR_ALLOC_FAIL;
308
309 seq->buf_len = seq2->len;
310 seq->len = 0;
311 }
312 else if (seq->len + seq2->len > seq->buf_len)
313 {
314 int new_len = seq->buf_len;
315 while (new_len < seq->len + seq2->len)
316 new_len *= 2;
317
318 uint8_t *new_ptr = kest_realloc(seq->buf, new_len * sizeof(uint8_t));
319
320 if (!new_ptr) return ERR_ALLOC_FAIL;
321
322 seq->buf_len = new_len;
323 seq->buf = new_ptr;
324 }
325
326 for (int i = 0; i < seq2->len; i++)
327 seq->buf[seq->len + i] = seq2->buf[i];
328
329 seq->len += seq2->len;
330
331 return 0;
332}

334int kest_fpga_transfer_batch_send_careful(kest_fpga_transfer_batch batch)
335{
336 KEST_PRINTF_FORCE("Sending program batch...\n");
337
338 uint8_t flags;
339 kest_string str;
340 kest_string_init_reserved(&str, 512);
341
342 for (int i = 0; i < batch.len; i++)
343 {
344 kest_fpga_txrx(&batch.buf[i], &flags, 1);
345 if (!(flags & 0b10))
346 {

348 }
349 else
350 {
351 kest_string_appendf(&str, "Sent 0x%02x, received %s\n", batch.buf[i], binary_print_8(flags));
352 }
353 if (i && (i % 128) == 0)
354 {
355 kest_puts(str);
356 kest_string_drain(&str);
357 }
358 }
359
360 kest_puts(str);

362 kest_string_destroy(&str);

490int kest_fpga_decode_status_flags(kest_fpga_status_flags *flags, uint8_t byte)
491{
492 if (!flags)
493 return ERR_NULL_PTR;
494
495 flags->initialised = !!(byte & (1 << 0));
496 flags->listening = !!(byte & (1 << 1));
497 flags->timeout = !!(byte & (1 << 2));
498 flags->programming = !!(byte & (1 << 3));
499 flags->bad = !!(byte & (1 << 4));
500 flags->data_ready = !!(byte & (1 << 5));
501 flags->cmd_err = !!(byte & (1 << 6));
502 flags->swapping = !!(byte & (1 << 7));
503
504 return NO_ERROR;
505}

560int64_t kest_fpga_req_data(int req, int n_bytes, kest_fpga_status_flags *flags)
561{
562 if (!flags) return -1;
563
564 kest_fpga_send_byte_get_flags(COMMAND_READ, flags);
565 kest_fpga_send_byte_get_flags(req, flags);
566
567 if (flags->cmd_err)
568 return -2;
569
570 int tries = 16;
571
572 do {
573 kest_fpga_get_status_flags(flags);
574 if (flags->cmd_err) return -4;
575 #ifdef ENABLE_FREERTOS
576 if (!flags->data_ready)
577 vTaskDelay(1);
578 #endif
579 } while (!flags->data_ready && tries --> 0);
580
581 int64_t data = 0;
582
583 if (flags->data_ready)
584 {
585 kest_fpga_send_byte_get_flags(COMMAND_READOUT, flags);
586
587 if (flags->cmd_err)
588 return -6;
589
590 for (int i = 0; i < n_bytes; i++)
591 data = (data << 8) | kest_fpga_readout();
592 }
593 else
594 {
595 data = -5;
596 }
597
598 kest_fpga_send_byte_get_flags(COMMAND_CLEAR_CMD_ERR_FLAG, flags);
599
600 return data;
601}
