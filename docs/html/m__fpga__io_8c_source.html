20int16_t float_to_q_nminus1(float x, int shift)
21{
22 int n = (M_FPGA_DATA_WIDTH - 1) - shift;
23
24 float scale = (float)(1 << n);
25
26 float max = (float)((1 << (M_FPGA_DATA_WIDTH - 1)) - 1) / scale;
27 float min = -(float)(1 << (M_FPGA_DATA_WIDTH - 1)) / scale;
28
29 if (x > max) x = max;
30 if (x < min) x = min;
31
32 return (m_fpga_sample_t)lrintf(x * scale);
33}

246 new_len *= 2;

253 seq->buf = new_ptr;
